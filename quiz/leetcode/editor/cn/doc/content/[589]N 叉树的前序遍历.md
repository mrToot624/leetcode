#include <vector>

class Solution {
public:
    std::vector<int> preorder(Node* root) {
        traverse(root);
        return res;
    }

private:
    std::vector<int> res;

    void traverse(Node* root) {
        if (root == nullptr) {
            return;
        }
        // 前序遍历位置
        res.push_back(root->val);
        for (Node* child : root->children) {
            traverse(child);
        }
        // 后序遍历位置
    }
};
